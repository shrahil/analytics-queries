SELECT "ta"."testId", "ta"."id", "ta"."userId", "ta"."result"->>'totalMarks' as TotalMarks, "uc"."expiryAt", "uc"."courseId", "uc"."trial", "uc"."courseOfferId" FROM "TestAttempt" "ta" INNER JOIN "UserCourse" "uc" ON "uc"."userId" = "ta"."userId" WHERE "uc"."expiryAt" > now() AND "uc"."trial" = FALSE AND "ta"."testId" > 2773522 AND "ta"."testId" < 2773529 AND "ta"."completed" = true and "ta"."finishedAt" <= '2023-08-24 04:30:00';
