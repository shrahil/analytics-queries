SELECT "ta"."testId", "ta"."id", "ta"."userId", "ta"."result"->>'totalMarks' as TotalMarks, "up"."phone", "up"."displayName", "u"."email" FROM "TestAttempt" "ta" INNER JOIN "UserProfile" "up"  ON "ta"."userId" = "up"."userId" INNER JOIN "User" "u" ON "up"."userId" = "u"."id" WHERE "ta"."testId" > 2773517 AND "ta"."testId" < 2773523 AND "ta"."completed" = true and "ta"."finishedAt" <= '2023-08-24 04:30:00';
